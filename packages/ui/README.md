# @gaia/ui — Shared component library

The shared shadcn/ui component package consumed by all apps in this monorepo.

## Adding components

```bash
pnpm dlx shadcn@latest add badge -c apps/storybook
```

Components are written to `packages/ui/src/components/`.

```tsx
import { Badge } from "@gaia/ui/components/badge"
```

## Setting up a new app

### Vite + React

1. **Create the app directory**

```bash
mkdir -p apps/my-app/src
```

2. **Create `apps/my-app/package.json`** with these dependencies:

```json
{
  "name": "my-app",
  "private": true,
  "type": "module",
  "scripts": {
    "dev": "vite",
    "build": "tsc -b && vite build",
    "typecheck": "tsc --noEmit"
  },
  "dependencies": {
    "@gaia/ui": "workspace:*",
    "react": "^19",
    "react-dom": "^19"
  },
  "devDependencies": {
    "@tailwindcss/vite": "^4",
    "@types/react": "^19",
    "@types/react-dom": "^19",
    "@vitejs/plugin-react": "^6",
    "typescript": "~6",
    "vite": "^8"
  }
}
```

3. **Create `apps/my-app/components.json`** — tells shadcn where to write components:

```json
{
  "$schema": "https://ui.shadcn.com/schema.json",
  "style": "base-nova",
  "rsc": false,
  "tsx": true,
  "tailwind": {
    "config": "",
    "css": "../../packages/ui/src/styles/globals.css",
    "baseColor": "neutral",
    "cssVariables": true
  },
  "iconLibrary": "lucide",
  "aliases": {
    "components": "@/components",
    "hooks": "@/hooks",
    "lib": "@/lib",
    "utils": "@gaia/ui/lib/utils",
    "ui": "@gaia/ui/components"
  },
  "rtl": false
}
```

4. **Create `apps/my-app/vite.config.ts`** — add `@tailwindcss/vite` plugin:

```ts
import tailwindcss from "@tailwindcss/vite"
import react from "@vitejs/plugin-react"
import { defineConfig } from "vite"
import path from "path"

export default defineConfig({
  plugins: [react(), tailwindcss()],
  resolve: {
    alias: {
      "@": path.resolve(__dirname, "./src"),
    },
  },
})
```

5. **Create `apps/my-app/tsconfig.app.json`** — set `@gaia/ui/*` path alias:

```json
{
  "compilerOptions": {
    "target": "es2023",
    "lib": ["ES2023", "DOM"],
    "module": "esnext",
    "types": ["vite/client"],
    "moduleResolution": "bundler",
    "allowImportingTsExtensions": true,
    "verbatimModuleSyntax": true,
    "noEmit": true,
    "jsx": "react-jsx",
    "erasableSyntaxOnly": true,
    "strict": true,
    "noUnusedLocals": true,
    "noUnusedParameters": true,
    "paths": {
      "@/*": ["./src/*"],
      "@gaia/ui/*": ["../../packages/ui/src/*"]
    }
  },
  "include": ["src"]
}
```

6. **Import `@gaia/ui/globals.css`** in your entry point:

```tsx
// src/main.tsx
import "@gaia/ui/globals.css"
```

Run `pnpm install` and you are done. The remaining boilerplate (`tsconfig.json`, `tsconfig.node.json`, `index.html`) are Framework-generated - same as what `create-vite` scaffolds.

### Next.js (App Router)

1. **Create the app directory** and install dependencies:

```bash
mkdir -p apps/my-next-app/app
```

2. **Create `apps/my-next-app/package.json`**:

```json
{
  "name": "my-next-app",
  "private": true,
  "scripts": {
    "dev": "next dev",
    "build": "next build",
    "typecheck": "tsc --noEmit"
  },
  "dependencies": {
    "@gaia/ui": "workspace:*",
    "next": "^15",
    "react": "^19",
    "react-dom": "^19"
  },
  "devDependencies": {
    "@tailwindcss/postcss": "^4",
    "@types/node": "^24",
    "@types/react": "^19",
    "@types/react-dom": "^19",
    "postcss": "^8",
    "tailwindcss": "^4",
    "typescript": "~6"
  }
}
```

3. **Create `apps/my-next-app/next.config.ts`** — must transpile the shared package:

```ts
import type { NextConfig } from "next"

const nextConfig: NextConfig = {
  transpilePackages: ["@gaia/ui"],
}

export default nextConfig
```

4. **Create `apps/my-next-app/postcss.config.mjs`** — Tailwind v4 uses `@tailwindcss/postcss` instead of the old v3 plugin:

```mjs
/** @type {import('postcss-load-config').Config} */
const config = {
  plugins: {
    "@tailwindcss/postcss": {},
  },
}

export default config
```

5. **Create `apps/my-next-app/app/globals.css`** — import the shared CSS:

```css
@import "@gaia/ui/globals.css";
```

6. **Import the CSS in your root layout:**

```tsx
// app/layout.tsx
import "./globals.css"

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>{children}</body>
    </html>
  )
}
```

7. **Create `apps/my-next-app/components.json`** — same config as Vite (step 3 above).

Run `pnpm install` and you are done. The remaining files (`tsconfig.json`, `next-env.d.ts`, `app/page.tsx`) are auto-generated by Next.js.

---

### Notes

- **`@gaia/ui/globals.css`** imports Tailwind v4 (`@import "tailwindcss"`) plus shadcn tokens and the Inter font. Every app must import this CSS entry point exactly once.
- **`@source` directives** in `globals.css` already scan `../../../apps/**/*.{ts,tsx}`, so Tailwind picks up all app source files without extra config.
- **TypeScript path aliases** (`@gaia/ui/*` → `../../packages/ui/src/*`) are needed so TS resolves the shared package during development. At runtime, the `workspace:*` symlink handles resolution.
- **Turbo** discovers `apps/*` automatically from `pnpm-workspace.yaml` — no turbo.json changes needed for new apps. For Next.js, add `"next build"` to turbo.json if you want orchestrated builds.
